OBTENER TODAS LAS INSTANCIAS DE LA CLASE COMIC 

SELECT DISTINCT ?comic ?comicLabel
WHERE {
#occupation (P106) must be Formula 1 driver (Q10841764)
?comic wdt:P31 wd:Q1004.
SERVICE wikibase:label { bd:serviceParam wikibase:language "es". }
}

OBTENER LAS PROPIEDADES DE LOS COMICS

SELECT DISTINCT ?p ?pLabel
WHERE {
#Instancia de (P31) que son comics (Q1004)
?comic wdt:P31 wd:Q1004;
?p ?x.
#to obtain property labels:
?property wikibase:directClaim ?p
#to obtain labels in spanish:
SERVICE wikibase:label { bd:serviceParam wikibase:language "es". }
}
